🔧 Python Sketch: Evolving a Periodic Implicit Surface

python
import torch
import torch.nn as nn
import numpy as np

# Device setup
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

# Parameters
grid_size = 32  # resolution of the domain
fourier_order = 3  # controls number of sine components
n_samples = 10000  # number of samples for Monte Carlo integral

# Create uniform sample points in the unit cube
xyz = torch.rand(n_samples, 3, device=device, requires_grad=True)  # shape [n_samples, 3]

# Define the Fourier-based implicit surface function
class FourierSurface(nn.Module):
    def __init__(self, order):
        super().__init__()
        self.order = order
        self.coeffs = nn.Parameter(torch.randn((order, order, order), device=device) * 0.1)

    def forward(self, xyz):
        x, y, z = xyz[:, 0], xyz[:, 1], xyz[:, 2]
        f = torch.zeros_like(x)
        for i in range(self.order):
            for j in range(self.order):
                for k in range(self.order):
                    a = self.coeffs[i, j, k]
                    f += a * torch.sin(2 * np.pi * (i+1) * x) * \
                             torch.sin(2 * np.pi * (j+1) * y) * \
                             torch.sin(2 * np.pi * (k+1) * z)
        return f

# Surface model
surface = FourierSurface(fourier_order).to(device)

# Optimizer
optimizer = torch.optim.Adam(surface.parameters(), lr=1e-2)

# Training loop
for step in range(1000):
    optimizer.zero_grad()

    f_val = surface(xyz)

    # Compute gradient ∇f
    grad_f = torch.autograd.grad(f_val, xyz, grad_outputs=torch.ones_like(f_val),
                                  create_graph=True, retain_graph=True)[0]
    
    # Compute |∇f|
    norm_grad_f = grad_f.norm(dim=1, keepdim=True) + 1e-8

    # Compute divergence of normalized gradient: ∇ ⋅ (∇f / |∇f|)
    grad_normalized = grad_f / norm_grad_f
    div = 0
    for i in range(3):
        div += torch.autograd.grad(grad_normalized[:, i], xyz, 
                                   grad_outputs=torch.ones_like(f_val),
                                   create_graph=True, retain_graph=True)[0][:, i]

    mean_curvature = div
    energy = ((mean_curvature ** 2) * torch.exp(-100 * f_val ** 2)).mean()  # narrow-band integral
    energy.backward()
    optimizer.step()

    if step % 100 == 0:
        print(f"Step {step}: Energy = {energy.item():.6f}")

  • This approach minimizes ∫H2δ(f)∣∇f∣ dx\int H^2 \delta(f) |\nabla f| \, dx∫H2δ(f)∣∇f∣dx, giving us a proxy for total curvature energy.

🧠 The Goal

We want to compute the surface integral:

where HHH is the mean curvature of the surface defined implicitly by f(x,y,z)=0f(x, y, z) = 0f(x,y,z)=0.

🔄 What We Do Instead

Because we don't have an explicit surface mesh during optimization, we approximate this surface integral in volumetric space using the co-area formula and a delta-function approximation:

∫f=0H2 dA≈∫R3H2 δ(f) ∣∇f∣ dx dy dz\int_{f=0} H^2 \, dA \approx \int_{\mathbb{R}^3} H^2 \, \delta(f) \, |\nabla f| \, dx\,dy\,dz∫f=0​H2dA≈∫R3​H2δ(f)∣∇f∣dxdydz

This lets us compute everything in 3D space without needing to extract the surface at each step.

💡 In the Code

In the training loop, we have this line:

  • mean_curvature ** 2 → H2H^2H2

  • f_val is the value of the function f(x,y,z)f(x, y, z)f(x,y,z)

  • exp(-100 * f_val**2) → a smoothed delta function centered at f=0f = 0f=0

    • This approximates δ(f)\delta(f)δ(f)

  • The .mean() over random samples approximates the volume integral:

    Ex∼Uniform[H2 δ(f) ∣∇f∣]≈Integral\mathbb{E}_{x \sim \text{Uniform}} \left[ H^2 \, \delta(f) \, |\nabla f| \right] \approx \text{Integral}Ex∼Uniform​[H2δ(f)∣∇f∣]≈Integral

import pyvista as pv
import vtk
from vtk.util import numpy_support

def export_scalar_fields_to_vti(f_model, resolution=64, filename="fields.vti"):
    # Create a regular grid
    x = np.linspace(0, 1, resolution)
    y = np.linspace(0, 1, resolution)
    z = np.linspace(0, 1, resolution)
    xx, yy, zz = np.meshgrid(x, y, z, indexing='ij')
    grid_points = torch.tensor(np.stack([xx, yy, zz], axis=-1), dtype=torch.float32).to(device)
    grid_flat = grid_points.view(-1, 3).clone().detach().requires_grad_(True)

    with torch.no_grad():
        f_vals = f_model(grid_flat)
    f_vals.requires_grad_(True)

    # Compute grad(f)
    grad_f = torch.autograd.grad(f_vals, grid_flat, grad_outputs=torch.ones_like(f_vals),
                                 create_graph=True, retain_graph=True)[0]
    norm_grad_f = grad_f.norm(dim=1, keepdim=True) + 1e-8
    grad_normalized = grad_f / norm_grad_f

    # Compute divergence of normalized gradient = mean curvature
    mean_curv = 0
    for i in range(3):
        mean_curv += torch.autograd.grad(grad_normalized[:, i], grid_flat,
                                         grad_outputs=torch.ones_like(f_vals),
                                         create_graph=True, retain_graph=True)[0][:, i]

    f_vals = f_vals.detach().cpu().numpy().reshape((resolution, resolution, resolution))
    mean_curv = mean_curv.detach().cpu().numpy().reshape((resolution, resolution, resolution))

    # Create VTK image data for VisIt
    image = vtk.vtkImageData()
    image.SetDimensions(resolution, resolution, resolution)
    image.SetSpacing(1.0 / resolution, 1.0 / resolution, 1.0 / resolution)
    image.SetOrigin(0, 0, 0)

    f_array = numpy_support.numpy_to_vtk(num_array=f_vals.ravel(order='F'), deep=True)
    f_array.SetName("f")
    image.GetPointData().AddArray(f_array)

    h_array = numpy_support.numpy_to_vtk(num_array=mean_curv.ravel(order='F'), deep=True)
    h_array.SetName("H")
    image.GetPointData().AddArray(h_array)

    writer = vtk.vtkXMLImageDataWriter()
    writer.SetFileName(filename)
    writer.SetInputData(image)
    writer.Write()
    print(f"✅ Exported scalar fields to {filename}")

This will write the sample points (xyz) and their corresponding mean curvature to a VTK PolyData file (.vtp), which VisIt can overlay on top of your .vti surface plot.

python
def export_sample_points_with_curvature(xyz, mean_curvature, filename="sample_points.vtp"):
    import vtk
    from vtk.util import numpy_support

    xyz_np = xyz.detach().cpu().numpy()
    H_np = mean_curvature.detach().cpu().numpy()

    # Create VTK Points object
    points = vtk.vtkPoints()
    points.SetData(numpy_support.numpy_to_vtk(xyz_np, deep=True))

    # Create VTK PolyData to hold points
    polydata = vtk.vtkPolyData()
    polydata.SetPoints(points)

    # Add mean curvature as a point scalar
    h_array = numpy_support.numpy_to_vtk(H_np, deep=True)
    h_array.SetName("H")
    polydata.GetPointData().AddArray(h_array)

    # Write to VTP file
    writer = vtk.vtkXMLPolyDataWriter()
    writer.SetFileName(filename)
    writer.SetInputData(polydata)
    writer.Write()
    print(f"✅ Exported sample points with curvature to {filename}")
